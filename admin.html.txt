<script>
    // ==================== ЗАГРУЗКА ДАННЫХ ====================
    function loadOrders() {
        const orders = JSON.parse(localStorage.getItem('uroboros_orders')) || [];
        return orders;
    }

    function saveOrders(orders) {
        localStorage.setItem('uroboros_orders', JSON.stringify(orders));
    }

    function loadProducts() {
        const defaultProducts = [
            { id: 1, name: "Excalibur силуэт", price: 2490, category: "warframe", image: "🎮", sizes: ["XS","S","M","L","XL"] },
            { id: 2, name: "Волт — скорость", price: 2490, category: "warframe", image: "⚡", sizes: ["XS","S","M","L","XL"] },
            { id: 3, name: "Лотос символ", price: 1990, category: "warframe", image: "🌸", sizes: ["S","M","L"] },
            { id: 4, name: "Черный минимум", price: 1890, category: "silhouette", image: "🖤", sizes: ["XS","S","M","L","XL","XXL"] },
            { id: 5, name: "Серый оверсайз", price: 2190, category: "silhouette", image: "⬜", sizes: ["S","M","L","XL"] },
            { id: 6, name: "Рик и Морти — кринж", price: 2590, category: "meme", image: "🤡", sizes: ["S","M","L","XL"] },
            { id: 7, name: "Жиза / Пов", price: 1990, category: "meme", image: "🔥", sizes: ["XS","S","M","L"] },
            { id: 8, name: "Базовый свитшот", price: 2390, category: "stock", image: "📦", sizes: ["S","M","L","XL","XXL"] }
        ];
        const stored = localStorage.getItem('uroboros_products');
        if (stored) {
            return JSON.parse(stored);
        } else {
            localStorage.setItem('uroboros_products', JSON.stringify(defaultProducts));
            return defaultProducts;
        }
    }

    function saveProducts(products) {
        localStorage.setItem('uroboros_products', JSON.stringify(products));
        // Также обновляем глобальный массив в script.js (сайт)
        if (typeof window.updateExternalProducts === 'function') {
            window.updateExternalProducts(products);
        }
    }

    // ==================== ОТРИСОВКА ЗАКАЗОВ ====================
    function renderOrders(filterText = '') {
        const orders = loadOrders();
        const tbody = document.getElementById('orders-list');
        if (!tbody) return;
        
        let filtered = orders;
        if (filterText) {
            const lower = filterText.toLowerCase();
            filtered = orders.filter(o => 
                o.name?.toLowerCase().includes(lower) || 
                o.phone?.includes(lower) || 
                o.address?.toLowerCase().includes(lower)
            );
        }
        
        if (filtered.length === 0) {
            tbody.innerHTML = '<tr><td colspan="9" style="text-align:center;">Нет заказов</td></tr>';
            return;
        }
        
        tbody.innerHTML = filtered.map(order => `
            <tr>
                <td>${order.id || '—'}</td>
                <td>${new Date(order.date).toLocaleString()}</td>
                <td>${order.name || '—'}</td>
                <td>${order.phone || '—'}</td>
                <td>${order.address || '—'}</td>
                <td>${order.items?.map(i => `${i.name} x${i.quantity}`).join(', ') || '—'}</td>
                <td>${order.total || 0} ₽</td>
                <td><span class="status-badge ${order.status || 'new'}">${order.status || 'new'}</span></td>
                <td>
                    <button class="update-status" data-id="${order.id || order.date}">🔄 Статус</button>
                </td>
            </tr>
        `).join('');
        
        // Обработчики для кнопок изменения статуса
        document.querySelectorAll('.update-status').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const orderId = btn.dataset.id;
                const ordersList = loadOrders();
                const orderIndex = ordersList.findIndex(o => (o.id || o.date).toString() === orderId);
                if (orderIndex !== -1) {
                    const statuses = ['new', 'processing', 'shipped', 'completed'];
                    let next = statuses[(statuses.indexOf(ordersList[orderIndex].status || 'new') + 1) % statuses.length];
                    ordersList[orderIndex].status = next;
                    saveOrders(ordersList);
                    renderOrders(filterText);
                }
            });
        });
    }
    
    // ==================== ТОВАРЫ ====================
    function renderProducts() {
        const products = loadProducts();
        const tbody = document.getElementById('products-list');
        if (!tbody) return;
        
        tbody.innerHTML = products.map(p => `
            <tr>
                <td>${p.id}</td>
                <td>${p.name}</td>
                <td>${p.price} ₽</td>
                <td>${p.category}</td>
                <td>
                    <button class="delete-product" data-id="${p.id}">🗑️ Удалить</button>
                </td>
            </tr>
        `).join('');
        
        document.querySelectorAll('.delete-product').forEach(btn => {
            btn.addEventListener('click', () => {
                const id = parseInt(btn.dataset.id);
                let productsList = loadProducts();
                productsList = productsList.filter(p => p.id !== id);
                saveProducts(productsList);
                renderProducts();
            });
        });
    }
    
    // Добавление товара
    document.getElementById('add-product-btn')?.addEventListener('click', () => {
        const name = document.getElementById('new-name').value.trim();
        const price = parseInt(document.getElementById('new-price').value);
        const category = document.getElementById('new-category').value;
        const image = document.getElementById('new-image').value.trim() || '🆕';
        const sizesRaw = document.getElementById('new-sizes').value.trim();
        const sizes = sizesRaw ? sizesRaw.split(',').map(s => s.trim()) : ['S','M','L'];
        
        if (!name || isNaN(price)) {
            alert('Заполните название и цену');
            return;
        }
        
        const products = loadProducts();
        const newId = Math.max(...products.map(p => p.id), 0) + 1;
        products.push({
            id: newId,
            name, price, category, image,
            sizes: sizes
        });
        saveProducts(products);
        renderProducts();
        
        // Очистка формы
        document.getElementById('new-name').value = '';
        document.getElementById('new-price').value = '';
        document.getElementById('new-image').value = '';
        document.getElementById('new-sizes').value = '';
        alert('✅ Товар добавлен! Обновите основную страницу, чтобы увидеть изменения.');
    });
    
    // ==================== СТАТИСТИКА ====================
    function renderStats() {
        const orders = loadOrders();
        const products = loadProducts();
        const totalOrders = orders.length;
        const totalRevenue = orders.reduce((sum, o) => sum + (o.total || 0), 0);
        const uniqueCustomers = new Set(orders.map(o => o.phone)).size;
        
        const statsContainer = document.getElementById('stats-container');
        if (statsContainer) {
            statsContainer.innerHTML = `
                <div class="stat-card"><div class="stat-number">${totalOrders}</div><div class="stat-label">Всего заказов</div></div>
                <div class="stat-card"><div class="stat-number">${totalRevenue} ₽</div><div class="stat-label">Выручка</div></div>
                <div class="stat-card"><div class="stat-number">${uniqueCustomers}</div><div class="stat-label">Уникальных клиентов</div></div>
                <div class="stat-card"><div class="stat-number">${products.length}</div><div class="stat-label">Товаров</div></div>
            `;
        }
    }
    
    // Экспорт заказов в CSV
    document.getElementById('export-orders-btn')?.addEventListener('click', () => {
        const orders = loadOrders();
        if (orders.length === 0) {
            alert('Нет заказов для экспорта');
            return;
        }
        let csv = "ID,Дата,Имя,Телефон,Адрес,Товары,Сумма,Статус\n";
        orders.forEach(o => {
            const itemsStr = o.items?.map(i => `${i.name} x${i.quantity}`).join('; ') || '';
            csv += `"${o.id || ''}","${o.date}","${o.name}","${o.phone}","${o.address}","${itemsStr}",${o.total},"${o.status || 'new'}"\n`;
        });
        const blob = new Blob([csv], {type: 'text/csv'});
        const a = document.createElement('a');
        const url = URL.createObjectURL(blob);
        a.href = url;
        a.download = `uroboros_orders_${new Date().toISOString().slice(0,19)}.csv`;
        a.click();
        URL.revokeObjectURL(url);
    });
    
    // ==================== ВКЛАДКИ ====================
    document.querySelectorAll('.tab-btn').forEach(btn => {
        btn.addEventListener('click', () => {
            document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
            btn.classList.add('active');
            const tabName = btn.dataset.tab;
            document.querySelectorAll('.tab-pane').forEach(pane => pane.classList.remove('active'));
            document.getElementById(`${tabName}-tab`).classList.add('active');
            if (tabName === 'orders') renderOrders();
            if (tabName === 'products') renderProducts();
            if (tabName === 'stats') renderStats();
        });
    });
    
    // Поиск по заказам
    const searchInput = document.getElementById('order-search');
    if (searchInput) {
        searchInput.addEventListener('input', (e) => renderOrders(e.target.value));
    }
    
    // Инициализация
    renderOrders();
    renderProducts();
    renderStats();
    
    // Сохраняем заказ при оформлении из index.html
    window.saveOrderToLocal = function(orderData) {
        const orders = loadOrders();
        const newOrder = {
            id: Date.now(),
            ...orderData,
            status: 'new',
            date: new Date().toISOString()
        };
        orders.unshift(newOrder);
        saveOrders(orders);
    };
    
    // Перехватываем заказы из основного сайта (через событие)
    window.addEventListener('storage', (e) => {
        if (e.key === 'uroboros_orders') {
            if (document.getElementById('orders-tab')?.classList.contains('active')) renderOrders();
            if (document.getElementById('stats-tab')?.classList.contains('active')) renderStats();
        }
    });
</script>
